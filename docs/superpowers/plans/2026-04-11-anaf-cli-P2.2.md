# P2.2 — UblService

> **Workstream:** P2.2 (Wave W4) — first of two.
> **Predecessors:** P1.7 (LookupService) + P2.1 (action model).
> **Master plan:** `docs/superpowers/plans/2026-04-11-anaf-cli-phases.md`

**Goal:** Resolve supplier from the active context, hydrate supplier + customer from `LookupService`, apply overrides from the `UblBuildAction`, map to SDK `InvoiceInput`, and build the XML via `UblBuilder`. No commander wiring (that's P2.3).

## Public API (FROZEN)

```ts
// cli/src/services/ublService.ts
import { UblBuilder } from 'anaf-ts-sdk';
import type { Party, InvoiceInput } from 'anaf-ts-sdk';
import type { UblBuildAction } from '../actions';
import type { LookupService } from './lookupService';
import type { ContextService } from '../state';

export interface UblServiceOptions {
  contextService: ContextService;
  lookupService: LookupService;
  builder?: UblBuilder;
}

export interface UblBuildResult {
  xml: string;
  invoice: InvoiceInput;
}

export class UblService {
  constructor(opts: UblServiceOptions);
  buildFromAction(action: UblBuildAction): Promise<UblBuildResult>;
}
```

**Error contract:** all failures are `CliError` with the appropriate category:
- `local_state` / `NO_CURRENT_CONTEXT` / `CONTEXT_NOT_FOUND` — propagated from `ContextService`
- `anaf_api` / `LOOKUP_FAILED` / `LOOKUP_NOT_FOUND` — propagated from `LookupService`
- `generic` / `UBL_BUILD_FAILED` — thrown when `UblBuilder` fails

## Known SDK quirk: `AnafCompanyData.address` is a single string

The SDK's lookup response returns the address as a single monolithic string (`"Bucuresti, Sector 1, Strada X nr 1"`), but the SDK's `Party.address` is a structured `Address` (`{ street, city, postalZone, county?, countryCode? }`). For v1:

- Map `AnafCompanyData.address` → `Address.street` (whole string).
- Map `AnafCompanyData.postalCode ?? ''` → `Address.postalZone`.
- Leave `Address.city` empty unless overridden. Empty city is technically invalid UBL but the SDK's builder accepts it; downstream users who need a clean XML will supply `--customer-city` / `--supplier-city` overrides.
- Default `countryCode` to `'RO'`.

This quirk is well-known — the ANAF public API doesn't decompose addresses. A smarter parser is v1.1.

## Party mapping

```ts
function companyToParty(company: AnafCompanyData, fallbackCui: string): Party {
  const cui = company.vatCode ?? fallbackCui.replace(/^RO/i, '');
  const vatNumber = company.scpTva ? `RO${cui}` : undefined;
  return {
    registrationName: company.name,
    companyId: cui,
    vatNumber,
    address: {
      street: company.address ?? '',
      city: '',            // overridable
      postalZone: company.postalCode ?? '',
      countryCode: 'RO',
    },
    telephone: company.contactPhone || undefined,
  };
}
```

Merging overrides on top of this party uses the existing `mergePartyOverride` from `cli/src/actions/overrideMerge.ts`.

## InvoiceInput mapping

```ts
{
  invoiceNumber: action.invoice.invoiceNumber,
  issueDate: action.invoice.issueDate,
  dueDate: action.invoice.dueDate ?? action.invoice.overrides?.dueDate,
  currency: action.invoice.currency ?? action.invoice.overrides?.currency ?? 'RON',
  note: action.invoice.note ?? action.invoice.overrides?.note,
  supplier,
  customer,
  lines: action.invoice.lines.map(toInvoiceLine),
  paymentIban: action.invoice.paymentIban ?? action.invoice.overrides?.paymentIban,
  isSupplierVatPayer: supplier.vatNumber !== undefined,
}
```

Where `toInvoiceLine` simply drops `action`-layer field names through to SDK `InvoiceLine` (description, quantity, unitPrice, taxPercent, unitCode).

```ts
import { UblBuilder } from 'anaf-ts-sdk';
import type {
  AnafCompanyData,
  InvoiceInput,
  InvoiceLine,
  Party,
} from 'anaf-ts-sdk';
import { CliError } from '../output/errors';
import type { UblBuildAction, InvoiceLineAction, PartyOverride } from '../actions';
import { mergePartyOverride } from '../actions/overrideMerge';
import type { ContextService } from '../state';
import type { LookupService } from './lookupService';

export interface UblServiceOptions {
  contextService: ContextService;
  lookupService: LookupService;
  builder?: UblBuilder;
}

export interface UblBuildResult {
  xml: string;
  invoice: InvoiceInput;
}

export class UblService {
  private readonly contextService: ContextService;
  private readonly lookupService: LookupService;
  private readonly builder: UblBuilder;

  constructor(opts: UblServiceOptions) {
    this.contextService = opts.contextService;
    this.lookupService = opts.lookupService;
    this.builder = opts.builder ?? new UblBuilder();
  }

  async buildFromAction(action: UblBuildAction): Promise<UblBuildResult> {
    const context = this.contextService.resolve(action.context);

    const [supplierLookup, customerLookup] = await Promise.all([
      this.lookupService.getCompany(context.companyCui),
      this.lookupService.getCompany(action.invoice.customerCui),
    ]);

    const supplier = mergePartyOverride(
      companyToParty(supplierLookup, context.companyCui),
      action.invoice.overrides?.supplier as PartyOverride | undefined,
    );
    const customer = mergePartyOverride(
      companyToParty(customerLookup, action.invoice.customerCui),
      action.invoice.overrides?.customer as PartyOverride | undefined,
    );

    const invoice: InvoiceInput = {
      invoiceNumber: action.invoice.invoiceNumber,
      issueDate: action.invoice.issueDate,
      dueDate: action.invoice.dueDate ?? action.invoice.overrides?.dueDate,
      currency: action.invoice.currency ?? action.invoice.overrides?.currency ?? 'RON',
      note: action.invoice.note ?? action.invoice.overrides?.note,
      supplier,
      customer,
      lines: action.invoice.lines.map(toInvoiceLine),
      paymentIban: action.invoice.paymentIban ?? action.invoice.overrides?.paymentIban,
      isSupplierVatPayer: supplier.vatNumber !== undefined,
    };

    let xml: string;
    try {
      xml = this.builder.generateInvoiceXml(invoice);
    } catch (cause) {
      throw new CliError({
        code: 'UBL_BUILD_FAILED',
        message: `Failed to build UBL XML: ${(cause as Error).message}`,
        category: 'generic',
        details: { invoiceNumber: invoice.invoiceNumber },
      });
    }

    return { xml, invoice };
  }
}

function companyToParty(company: AnafCompanyData, fallbackCui: string): Party {
  const cui = company.vatCode ?? fallbackCui.replace(/^RO/i, '');
  const vatNumber = company.scpTva ? `RO${cui}` : undefined;
  return {
    registrationName: company.name,
    companyId: cui,
    vatNumber,
    address: {
      street: company.address ?? '',
      city: '',
      postalZone: company.postalCode ?? '',
      countryCode: 'RO',
    },
    telephone: company.contactPhone || undefined,
  };
}

function toInvoiceLine(line: InvoiceLineAction): InvoiceLine {
  const out: InvoiceLine = {
    description: line.description,
    quantity: line.quantity,
    unitPrice: line.unitPrice,
    taxPercent: line.taxPercent,
  };
  if (line.unitCode) out.unitCode = line.unitCode;
  return out;
}
```

## Handoff

- **P2.3** (`ubl build` command) calls `deps.services.ublService.buildFromAction(action)` and writes `result.xml` through the output layer.
- **P2.5** (`efactura upload` command with `--from-ubl-flags` or `source.ublBuild`) delegates to `ublService.buildFromAction` before uploading.
