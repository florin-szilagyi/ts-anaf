# P1.6 — `ctx` commands

> **Workstream:** P1.6 (Wave W2) — first of three to land in W2.
> **Predecessors:** P1.1, P1.2 (+ fix), P1.3, P1.4 — all merged on `feat/anaf-cli`.

**Goal:** Implement the `ctx ls`, `ctx current`, `ctx use`, `ctx add`, `ctx rm`, `ctx rename` commands by replacing the `notImplemented` stubs in `cli/src/commands/groups/ctx.ts` with handlers that call the existing `ContextService` from `cli/src/state/`. Also establish the **service-registry merge pattern** that all W2+ workstreams will reuse.

**Architecture:** Handlers are pure functions exported from the group file: `(args, opts, deps) => Promise<void>`. They read state via `deps.services.contextService` and emit through `deps.output`. The commander `.action()` callback is a thin adapter that converts commander's call shape into a handler call. This makes handlers unit-testable without parsing.

## Service-registry merge pattern (FROZEN — all W2+ workstreams use this)

`ServiceRegistry` becomes a record of REQUIRED service instances. `RunProgramOptions.services` stays `Partial<ServiceRegistry>`. `runProgram` builds the full registry by filling missing fields with default-constructed services:

```ts
// cli/src/commands/buildProgram.ts
import type { ContextService } from '../state';
export interface ServiceRegistry {
  contextService: ContextService;
  // P1.7 adds: lookupService: LookupService;
  // P1.5 adds: tokenStore: TokenStore; authService: AuthService;
  // P2.4 adds: efacturaService: EfacturaService;
  // P2.2 adds: ublService: UblService;
  // P3.1 adds: manifestService: ManifestService;
}
```

```ts
// cli/src/commands/runProgram.ts  (inside the try block, after building output)
const services: ServiceRegistry = {
  contextService: options.services?.contextService ?? new ContextService(),
  // P1.7+ append more lines here
};
```

Tests inject by passing `services: { contextService: mockedInstance }` and `runProgram` will fill any missing services with defaults.

**Why this shape:** every command handler can safely access `deps.services.contextService` without an existence check. Test code can override one field at a time.

## Handler contract (FROZEN — all command groups use this)

```ts
// shared shape; not a literal interface, just a convention
export type HandlerArgs<TArgs extends Record<string, unknown> = Record<string, never>> = {
  positional: TArgs;          // commander positional args, named
  options: Record<string, unknown>;  // raw commander options (handler casts)
  deps: CommandDeps;
};
```

For P1.6, each handler is a named export from `cli/src/commands/groups/ctx.ts`. Example:

```ts
export async function ctxLs(deps: CommandDeps): Promise<void> {
  const contexts = deps.services.contextService.list();
  const current = (() => {
    try { return deps.services.contextService.resolve().name; } catch { return undefined; }
  })();
  deps.output && /* ... renderSuccess with a human formatter ... */;
}
```

The `.action()` callback in `registerCtx` calls these by name.

### Task 4: Implement handlers

- [ ] **Step 1**: Replace `cli/src/commands/groups/ctx.ts`. Keep `registerCtx`, but bind real handlers.

```ts
import type { Command } from 'commander';
import type { CommandDeps } from '../buildProgram';
import { CliError } from '../../output/errors';
import { renderSuccess } from '../../output';
import type { Context, Environment } from '../../state';

interface CtxAddOptions {
  name?: string;
  cui?: string;
  clientId?: string;
  redirectUri?: string;
  env?: string;
}

export async function ctxLs(deps: CommandDeps): Promise<void> {
  const contexts = deps.services.contextService.list();
  let current: string | undefined;
  try {
    current = deps.services.contextService.resolve().name;
  } catch {
    current = undefined;
  }
  const data = {
    current,
    contexts: contexts.map((c) => ({
      name: c.name,
      companyCui: c.companyCui,
      environment: c.environment,
      isCurrent: c.name === current,
    })),
  };
  renderSuccess(deps.output, data, (d) => {
    if (d.contexts.length === 0) return '(no contexts)';
    return d.contexts
      .map((c) => `${c.isCurrent ? '*' : ' '} ${c.name}\t${c.environment}\t${c.companyCui}`)
      .join('\n');
  });
}

export async function ctxCurrent(deps: CommandDeps): Promise<void> {
  const ctx = deps.services.contextService.resolve();
  renderSuccess(deps.output, { name: ctx.name, companyCui: ctx.companyCui, environment: ctx.environment }, (d) => d.name);
}

export async function ctxUse(deps: CommandDeps, name: string): Promise<void> {
  // Throws CONTEXT_NOT_FOUND if the named context does not exist.
  deps.services.contextService.get(name);
  // ConfigStore is reachable through ContextService's internal state via setCurrent helper.
  // Add a setCurrent method on ContextService if not already present (P1.4 added one indirectly
  // via remove/rename; here we go through the underlying ConfigStore).
  const config = (deps.services.contextService as unknown as { config: { setCurrentContext(n: string): void } }).config;
  // Cleaner: expose a public setCurrent on ContextService. Do this in Step 1b below.
  config.setCurrentContext(name);
  renderSuccess(deps.output, { current: name }, (d) => `current context: ${d.current}`);
}

export async function ctxAdd(deps: CommandDeps, opts: CtxAddOptions): Promise<void> {
  const required: Array<keyof CtxAddOptions> = ['name', 'cui', 'clientId', 'redirectUri'];
  const missing = required.filter((k) => !opts[k]);
  if (missing.length > 0) {
    throw new CliError({
      code: 'BAD_USAGE',
      message: `ctx add: missing required options: ${missing.map((k) => `--${kebab(k)}`).join(', ')}`,
      category: 'user_input',
      details: { missing },
    });
  }
  const env = opts.env ?? 'prod';
  if (env !== 'test' && env !== 'prod') {
    throw new CliError({
      code: 'BAD_USAGE',
      message: `ctx add: --env must be 'test' or 'prod' (got "${env}")`,
      category: 'user_input',
      details: { env },
    });
  }
  const ctx: Context = {
    name: opts.name!,
    companyCui: opts.cui!,
    environment: env as Environment,
    auth: {
      clientId: opts.clientId!,
      redirectUri: opts.redirectUri!,
    },
  };
  const created = deps.services.contextService.add(ctx);
  renderSuccess(
    deps.output,
    { name: created.name, environment: created.environment, companyCui: created.companyCui },
    (d) => `added context: ${d.name} (${d.environment}, ${d.companyCui})`,
  );
}

export async function ctxRm(deps: CommandDeps, name: string): Promise<void> {
  deps.services.contextService.remove(name);
  renderSuccess(deps.output, { removed: name }, (d) => `removed: ${d.removed}`);
}

export async function ctxRename(deps: CommandDeps, oldName: string, newName: string): Promise<void> {
  const renamed = deps.services.contextService.rename(oldName, newName);
  renderSuccess(deps.output, { from: oldName, to: renamed.name }, (d) => `renamed: ${d.from} → ${d.to}`);
}

function kebab(s: string): string {
  return s.replace(/[A-Z]/g, (m) => `-${m.toLowerCase()}`);
}

export function registerCtx(parent: Command, deps: CommandDeps): void {
  const ctx = parent.command('ctx').description('Manage company contexts');

  ctx
    .command('ls')
    .description('List all configured contexts')
    .action(() => ctxLs(deps));

  ctx
    .command('current')
    .description('Print the current context')
    .action(() => ctxCurrent(deps));

  ctx
    .command('use <name>')
    .description('Set the current context')
    .action((name: string) => ctxUse(deps, name));

  ctx
    .command('add')
    .description('Add a new context')
    .option('--name <name>', 'context name')
    .option('--cui <cui>', 'company VAT number (CUI)')
    .option('--client-id <id>', 'OAuth client id')
    .option('--redirect-uri <uri>', 'OAuth redirect uri')
    .option('--env <env>', 'environment: test|prod', 'prod')
    .action((opts: CtxAddOptions) => ctxAdd(deps, opts));

  ctx
    .command('rm <name>')
    .description('Remove a context (and its token file)')
    .action((name: string) => ctxRm(deps, name));

  ctx
    .command('rename <oldName> <newName>')
    .description('Rename a context')
    .action((oldName: string, newName: string) => ctxRename(deps, oldName, newName));
}
```
