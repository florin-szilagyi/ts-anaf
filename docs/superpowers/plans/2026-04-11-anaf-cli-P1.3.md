# P1.3 — Output layer

> **Workstream:** P1.3 (Wave W1) — first of three to land in W1.
> **Predecessor:** P1.1 (merged as `bdd829d`).
> **Master plan:** `docs/superpowers/plans/2026-04-11-anaf-cli-phases.md`

**Goal:** Build the shared output layer the CLI uses everywhere: text + JSON renderers, error envelope with category-mapped exit codes, and stdout/stderr discipline. This module is consumed by every command in P1.2/P1.5/P1.6/P1.7/P2.x/P3.x.

**Architecture:** A pure module under `cli/src/output/` with no dependencies on commander, fs, or any SDK. Inputs are plain objects; outputs are written through an injected `WriteStreams` interface (default = `process.stdout` / `process.stderr`). This keeps the layer trivially testable without spawning processes.

## Public API (FROZEN — downstream depends on this)

```ts
// cli/src/output/index.ts
export type OutputFormat = 'text' | 'json';

export interface WriteStreams {
  stdout: NodeJS.WritableStream;
  stderr: NodeJS.WritableStream;
}

export interface OutputContext {
  format: OutputFormat;
  streams: WriteStreams;
}

export interface SuccessEnvelope<T> {
  success: true;
  data: T;
}

export interface ErrorEnvelope {
  success: false;
  error: {
    code: string;
    message: string;
    details?: Record<string, unknown>;
  };
}

export type ResultEnvelope<T> = SuccessEnvelope<T> | ErrorEnvelope;

// Exit codes (§15.2 of design doc — initial set; full hardening in P3.5).
export const EXIT_CODES = {
  SUCCESS: 0,
  GENERIC_FAILURE: 1,
  USER_INPUT: 2,
  AUTH: 3,
  ANAF_API: 4,
  LOCAL_STATE: 5,
} as const;
export type ExitCode = (typeof EXIT_CODES)[keyof typeof EXIT_CODES];

// Error categories used by CliError; mapped 1:1 to exit codes by errorToExit().
export type ErrorCategory =
  | 'generic'
  | 'user_input'
  | 'auth'
  | 'anaf_api'
  | 'local_state';

export class CliError extends Error {
  readonly code: string;
  readonly category: ErrorCategory;
  readonly details?: Record<string, unknown>;
  constructor(args: {
    code: string;
    message: string;
    category: ErrorCategory;
    details?: Record<string, unknown>;
  });
}

export function errorToExit(error: CliError | Error): ExitCode;

export function defaultStreams(): WriteStreams;
export function makeOutputContext(args: { format?: OutputFormat; streams?: WriteStreams }): OutputContext;

export function renderSuccess<T>(ctx: OutputContext, data: T, humanText?: (data: T) => string): void;
export function renderError(ctx: OutputContext, error: CliError | Error): void;

// Binary helper for XML/PDF — writes raw bytes to stdout (or to a file if path is set).
export function writeBinary(ctx: OutputContext, bytes: Uint8Array, opts?: { path?: string }): void;
```

**Contract for downstream callers:**
- **Never** call `console.log` / `console.error` directly. Always go through `renderSuccess` / `renderError`.
- Diagnostics and progress messages go to `ctx.streams.stderr` only. Stdout is reserved for the *result*.
- JSON mode wraps the result in `ResultEnvelope`. Text mode prints the human function's output verbatim.
- `errorToExit` is the only place that maps category → exit code. Process exit happens in the bin entrypoint, not in the renderers.

import { exitCodeForCategory, EXIT_CODES, type ExitCode } from './exitCodes';
import type { ErrorCategory } from './types';

export interface CliErrorArgs {
  code: string;
  message: string;
  category: ErrorCategory;
  details?: Record<string, unknown>;
}

export class CliError extends Error {
  readonly code: string;
  readonly category: ErrorCategory;
  readonly details?: Record<string, unknown>;

  constructor(args: CliErrorArgs) {
    super(args.message);
    this.name = 'CliError';
    this.code = args.code;
    this.category = args.category;
    this.details = args.details;
  }
}

export function errorToExit(error: CliError | Error): ExitCode {
  if (error instanceof CliError) {
    return exitCodeForCategory(error.category);
  }
  return EXIT_CODES.GENERIC_FAILURE;
}

import { CliError } from './errors';
import type { OutputContext } from './types';

export function renderSuccess<T>(
  ctx: OutputContext,
  data: T,
  humanText?: (data: T) => string,
): void {
  const line = humanText ? humanText(data) : JSON.stringify(data);
  ctx.streams.stdout.write(`${line}\n`);
}

export function renderError(ctx: OutputContext, error: CliError | Error): void {
  const code = error instanceof CliError ? error.code : 'GENERIC';
  ctx.streams.stderr.write(`${code}: ${error.message}\n`);

import { CliError } from './errors';
import type { ErrorEnvelope, OutputContext, SuccessEnvelope } from './types';

export function renderSuccess<T>(ctx: OutputContext, data: T): void {
  const envelope: SuccessEnvelope<T> = { success: true, data };
  ctx.streams.stdout.write(`${JSON.stringify(envelope)}\n`);
}

export function renderError(ctx: OutputContext, error: CliError | Error): void {
  const envelope: ErrorEnvelope = {
    success: false,
    error: {
      code: error instanceof CliError ? error.code : 'GENERIC',
      message: error.message,
      ...(error instanceof CliError && error.details !== undefined ? { details: error.details } : {}),
    },
  };
  ctx.streams.stderr.write(`${JSON.stringify(envelope)}\n`);
}
```

import fs from 'node:fs';
import path from 'node:path';
import type { OutputContext } from './types';

export function writeBinary(
  ctx: OutputContext,

}

export type {
  OutputFormat,
  WriteStreams,
  OutputContext,
  SuccessEnvelope,
  ErrorEnvelope,
  ResultEnvelope,
  ErrorCategory,
} from './types';

export { EXIT_CODES, type ExitCode, exitCodeForCategory } from './exitCodes';
export { CliError, errorToExit, type CliErrorArgs } from './errors';
export { defaultStreams, makeOutputContext } from './context';
export { writeBinary } from './binary';

import * as textRenderer from './text';
import * as jsonRenderer from './json';
import type { CliError as CliErrorType } from './errors';
import type { OutputContext } from './types';

export function renderSuccess<T>(
  ctx: OutputContext,
  data: T,
  humanText?: (data: T) => string,
): void {
  if (ctx.format === 'json') {
    jsonRenderer.renderSuccess(ctx, data);
    return;
  }
  textRenderer.renderSuccess(ctx, data, humanText);
}

export function renderError(ctx: OutputContext, error: CliErrorType | Error): void {
  if (ctx.format === 'json') {
    jsonRenderer.renderError(ctx, error);
    return;
  }
  textRenderer.renderError(ctx, error);
}
```
